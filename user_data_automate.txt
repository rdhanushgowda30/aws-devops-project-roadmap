#!/bin/bash
# EC2 Bootstrap Script for DevOps Project

# Update system
sudo apt update -y
sudo apt upgrade -y

# Install basic tools
sudo apt install -y git unzip curl docker.io

# Start and enable Docker
sudo systemctl start docker
sudo systemctl enable docker

# Install Terraform
TERRAFORM_VERSION="1.9.0"
curl -O https://releases.hashicorp.com/terraform/${TERRAFORM_VERSION}/terraform_${TERRAFORM_VERSION}_linux_amd64.zip
unzip terraform_${TERRAFORM_VERSION}_linux_amd64.zip
sudo mv terraform /usr/local/bin/
terraform -v

# Install AWS CLI v2
curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
unzip awscliv2.zip
sudo ./aws/install
aws --version

# Clone your GitHub repo automatically
cd /home/ubuntu
git clone https://github.com/rdhanushgowda30/aws-devops-project-roadmap.git

# Reminder for AWS configure
echo "Run 'aws configure' to set your credentials (Access Key, Secret Key, Region, Output)."
