#!/bin/bash
# EC2 Bootstrap Script for DevOps Project

# Log everything
exec > >(tee /var/log/user-data.log|logger -t user-data ) 2>&1

# Update system
apt-get update -y
apt-get upgrade -y

# Install basic tools
apt-get install -y git unzip curl docker.io

# Start and enable Docker
systemctl start docker
systemctl enable docker

# Install Terraform
TERRAFORM_VERSION="1.9.0"
curl -fsSL https://releases.hashicorp.com/terraform/${TERRAFORM_VERSION}/terraform_${TERRAFORM_VERSION}_linux_amd64.zip -o terraform.zip
unzip terraform.zip
mv terraform /usr/local/bin/
terraform -v

# Install AWS CLI v2
curl -fsSL "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o awscliv2.zip
unzip awscliv2.zip
./aws/install
aws --version

# Configure AWS CLI automatically
mkdir -p /home/ubuntu/.aws
cat <<EOT > /home/ubuntu/.aws/credentials
[default]
aws_access_key_id = AKIAY7LDBL6MKMEPI5E58055
aws_secret_access_key = WHanG0pbn8gTUm9Hp9K9D3GujOE7aCyI0a5pRiNidboss
EOT

cat <<EOT > /home/ubuntu/.aws/config
[default]
region = us-east-1
output = json
EOT

chown -R ubuntu:ubuntu /home/ubuntu/.aws

# Clone your GitHub repo
cd /home/ubuntu
git clone https://github.com/rdhanushgowda30/aws-devops-project-roadmap.git

# Run a sample Docker container
docker run -d -p 80:80 nginx
